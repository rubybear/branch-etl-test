CREATE TABLE users (uuid text primary key, gender text, email text, phone text, cell text, nat text)
CREATE TABLE dob (uuid text primary key, date timestamp, age int)
CREATE TABLE name (uuid text primary key, title text, first text, last text)
CREATE TABLE location (uuid text primary key, city text, state text, country text, postcode int )
CREATE TABLE street (uuid text primary key, number int, name text)
CREATE TABLE coordinates (uuid text primary key, latitude float, longitude float)
CREATE TABLE timezone (uuid text primary key, offset float, description text)
CREATE TABLE login (uuid text primary key, username text, password text, salt text, md5 text, shar1 text, sha256 text)
CREATE TABLE id (uuid text primary key, name text, value text)
CREATE TABLE registered (uuid text primary key, date date, age int)
CREATE TABLE picture (uuid text primary key, large text, medium text, thumbnail text)
